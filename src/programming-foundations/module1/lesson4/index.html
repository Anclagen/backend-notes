Task Part 1

  • 1. Load the data from the user (via prompt). You should get a number greater than 1 but less than 20. If the data from the user is incorrect, you should show them another prompt repeatedly until you get the correct number.
  • 2. Print all multiplicities of these positive integer numbers in the console that are less than 1000 and not multiplicities of 23. Which loop is best for our task?

The Code

        
    // 1
    let valueOne = 0;
    
    while (valueOne > 20 || valueOne < 1) {
      valueOne = parseFloat(prompt("Enter a number between 1 and 20"));
    }
    
    // 2
    function getMultiplicities(value) {
      for (let i = 1; i <= 1000; i++) {
        if (i % value === 0) {
          console.log(i);
        }
      }
    }
    
    getMultiplicities(23);

Task Part 2

  • 1. Get the input value from the user. The value should be between 0 and 100.
  • 2. Create two buttons and apply two functions to them on click: one will increase the stored value by 1, while the second function will decrease it.
  • 3. After every button click, show the current variable value in the console. After reaching either 0 or 100, instead of increasing/decreasing the value, communicate “Game over” in the console.

The Code

           
      // 1

      let valueTwo = "";

      while (valueTwo >= 100 || valueTwo <= 0 || isNaN(valueTwo)) {
        valueTwo = parseFloat(prompt("Enter a number between 0 and 100"));
      }

      // 2 + 3
      let gameOver = false;
      function gameOver() {
        gameOver = true;
        console.log("Game over");
        document.getElementById("gameOver").innerHTML = "Game over";
      }
      
      function increment() {
        if (valueTwo === 100 || gameOver === true) {
          gameOver();
          return;
        } else {
          valueTwo++;
          document.getElementById("result2").innerHTML = valueTwo;
          console.log(valueTwo);
        }
      }
      
      function decrement() {
        if (valueTwo === 0 || gameOver === true) {
          gameOver();
          return;
        } else {
          valueTwo--;
          document.getElementById("result2").innerHTML = valueTwo;
          console.log(valueTwo);
        }
      }
      
      // reset the game
      function reset() {
        gameOver = false;
        valueTwo = "";
        while (valueTwo >= 100 || valueTwo <= 0 || isNaN(valueTwo)) {
          valueTwo = parseFloat(prompt("Enter a number between 0 and 100"));
        }
        document.getElementById("result2").innerHTML = valueTwo;
        document.getElementById("gameOver").innerHTML = "";
      }
